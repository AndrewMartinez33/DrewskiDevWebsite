JavaScript Notes

    TYPES

    
              // number
              // string
              // boolean
              // undefined - a variable that is created but is not assigned a value
              // null - signifies an intentional absence of data
              // symbol
              // object

    TYPE OF KEYWORD

    
              typeof 5; // returns the data type, this returns 'number'

    VARIABLES

    
              var a = 0;    // function scope, no block scope, hoisted to top, stop using
              let a = 0;    // block scope
              const a = 0;  // block scope, does not change

    COERCION

    coercion is the process of converting a value from one type to another

    
              // foo is coerced to true
              if ('foo') {
                return;
              }

              // 5 is corced to a string, resulting in '5hi'
              5 + 'hi';

    DOUBLE AND TRIPLE EQUALS

    
              ==  // Loose, checks that both values are equal, allows type coercion
              === // Strict, checks that both values are equal, no type coercion

    TRUTHY and FALSEY VALUES

    
              // In JS there are 6 falsey values. They are coerced to false in a boolean expression
              // All other values are truthy and are coerced to true
              
              0;
              ('');
              null;
              undefined;
              false;
              NaN;

    NOT, OR, AND OPERATORS

    
              !     // Not Operator, flips a boolean value. !! double flips
              ||    // Or Operator, returns true if either condition is true
              &&    // And Operator, return true if both conditions are true

    IF/ ELSE IF/ ELSE

    
              if (conditionTrue) {
                return "dosomething";
              } else if (conditionTrue) {
                return "do something";
              } else {
                return "just do this";
              }

    SWITCH STATEMENTS

    
              //The switch expression is evaluated once.
              switch( expression ){
                case n:
                  code block;
                  break;
                case n:
                  //The value of the expression is compared with the values of each case.
                  code block;
                  break;
                case n:
                  //If there is a match, the associated block of code is executed.
                  code block;
                  break;
                default:
                  code block;
              }

    LOOPS

    
              // for loop
              for (let i = 0; i < 10; i++) {
                someCode;
              }

              // while loop
              while (condition) {
                someCode;
              }

              // do while loop
              do {
                someCode;
              } while (condition);

              // for in loop
              // interates over object's properties
              for (let key in object) {
                console.log(key);
              }

              // for of loop
              // iterates over object's values
              for (let value of object) {
                console.log(value);
              }

    Array.forEach(func)

    
              // Executes provided function for each item in an array
              array.forEach((item) => {
                someCode;
              })
